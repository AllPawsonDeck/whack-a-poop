<body>

    <h1>Whack-a-Poop</h1>

    <p>Click the 💩 to score points!</p>

    <div id="score">Score: 0</div>

    <div id="gameContainer"></div>

   

    <script>

        const gameContainer = document.getElementById('gameContainer');

        let score = 0;

        function createHole() {

            const hole = document.createElement('div');

            hole.classList.add('hole');

            gameContainer.appendChild(hole);

            return hole;

        }

        const holes = Array.from({ length: 9 }, createHole);

        function randomPoop() {

            const index = Math.floor(Math.random() * holes.length);

            holes.forEach(hole => hole.innerHTML = '');

            const poop = document.createElement('div');

            poop.innerHTML = '💩';

            poop.style.cursor = 'pointer';

            holes[index].appendChild(poop);

            poop.addEventListener('click', () => {

                score++;

                document.getElementById('score').innerText = `Score: ${score}`;

                holes[index].innerHTML = '';

            });

        }

        setInterval(randomPoop, 1000);

    </script>
